# Endpoints

# Get all recipe
GET '/api/recipes'

# Get a single recipe
 GET '/api/recipes/:id'
 
 # Create a new recipe
POST '/api/recipes'
# Edit a recipe
 PUT '/api/recipes/:id'

 # Delete a recipe
 DELETE '/api/recipes/:id'
